// Firestore Security Rules — copy ALL of firestore.rules into:
// Firebase Console → Firestore Database → Rules → Publish
//
// Without the users/ rules below, login fails with:
// "Missing or insufficient permissions"

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() {
      return request.auth != null;
    }

    function ownProfile() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function isApprovedAdmin() {
      return isSignedIn()
        && ownProfile().data.role == 'ADMIN'
        && ownProfile().data.accountStatus == 'approved';
    }

    function rejectedStatus(data) {
      return data.accountStatus == 'rejected' || data.accountStat == 'rejected';
    }

    function isRejectedSelfReapply() {
      return isSignedIn()
        && request.auth.uid == userId
        && rejectedStatus(resource.data)
        && request.resource.data.accountStatus == 'pending';
    }

    match /users/{userId} {
      allow read: if isSignedIn() && request.auth.uid == userId;
      allow read: if isApprovedAdmin();
      allow create: if isSignedIn() && request.auth.uid == userId;
      allow update: if isApprovedAdmin() || isRejectedSelfReapply();
      allow delete: if isApprovedAdmin();
    }

    match /researchProposals/{proposalId} {
      allow read, write: if true;
      match /tasks/{taskId} {
        allow read, write: if true;
      }
    }

    match /tasks/{taskId} {
      allow read, write: if true;
    }

    match /patents/{patentId} {
      allow read, write: if true;
    }

    match /articles/{articleId} {
      allow read, write: if true;
    }

    match /notifications/{notificationId} {
      allow read, write: if true;
    }

    match /researcherMessages/{messageId} {
      allow read, write: if true;
    }
  }
}
