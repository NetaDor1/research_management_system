// Firestore Security Rules for Research Management System
// Copy these rules to Firebase Console > Firestore Database > Rules
// NOTE: The app currently uses localStorage for roles (no Firebase Auth),
// so the rules below are intentionally permissive for development.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Research Proposals
    match /researchProposals/{proposalId} {
      allow read: if true;
      allow write: if true;
      
      // Tasks subcollection
      match /tasks/{taskId} {
        allow read, write: if true;
      }
    }
    
    // Global Tasks Collection (for calendar)
    match /tasks/{taskId} {
      allow read, write: if true;
    }
    
    // Patents
    match /patents/{patentId} {
      allow read, write: if true;
    }
    
    // Articles
    match /articles/{articleId} {
      allow read, write: if true;
    }

    // Notifications
    match /notifications/{notificationId} {
      allow read, write: if true;
    }

    // Researcher Messages (messages sent from researcher to authority)
    match /researcherMessages/{messageId} {
      allow read, write: if true;
    }
  }
}
